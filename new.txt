FROM eclipse-temurin:21-jdk  

# Install Maven + Git inside this single stage
RUN apt-get update && apt-get install -y maven git && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /sujith

# Clone the project
RUN git clone https://github.com/spring-projects/spring-petclinic.git

# Build the project
RUN cd /sujith/spring-petclinic && mvn clean package -DskipTests

# Copy built jar to a fixed path
RUN cp /sujith/spring-petclinic/target/*.jar /sujith/app.jar

# Expose application port
EXPOSE 8080

# Run the application
CMD ["java", "-jar", "/sujith/app.jar"]
